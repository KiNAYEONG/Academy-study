#root-context.xml
<!-- bean : 객체를 생성하는 태그 -->
<!-- HikariDataSource(Connection POOL을 만드는 역할을 한다) -->
<bean id="dataSource" class="com.zaxxer.hikari.HikariDataSource" destroy-method="close">
<constructor-arg ref="hikariConfig" /> <!--객체생성 참고해야하니깐!  -->
</bean>
   
<!-- Mapper interface들을 메모리에 올리는 방법(scan) -->
<mybatis-spring:scan base-package="kr.board.mapper"/><!-- 여기서 sql문 참고할거야 -->
   
<!-- BoardMapper interface의 구현클래스 생성
SqlSessionFactoryBean(SQL을 실행하는 API) -->
<bean class="org.mybatis.spring.SqlSessionFactoryBean">
<property name="dataSource" ref="dataSource" /> <!--커넥션이 데이터소스 담당 아니깐 참조! -->
</bean>
</beans>
